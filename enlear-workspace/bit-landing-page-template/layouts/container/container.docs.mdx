import { Container } from './container';
import { Button } from '@mui/material';

A component that acts as a container and wraps children.

### Component usage

```js
<Container>
  <div>Place any content</div>
</Container>
```

### Render a container with max width and styling

```js live
<Container maxWidth="lg" sx={{ p: 10 }}>
  <Button variant="contained">MaxWidth="lg"</Button>
</Container>
```
