import YouTubeIcon from '@mui/icons-material/YouTube';
import { Avatar, alpha } from '@mui/material';

A component that acts as a layout and places the app bar and footers that will wrap the content

### Component usage

```js
<LightTheme>
  <Main headerItems={[]} footerItems={[]}>
    {layoutChildren}
  </Main>
</LightTheme>
```
