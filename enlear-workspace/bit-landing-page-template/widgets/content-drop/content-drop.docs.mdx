import { ContentDrop } from './content-drop';
import InfoIcon from '@mui/icons-material/Info';
import { LightTheme } from '@enlear/bit-landing-page-template.themes.light-theme';

A component launches a popover with content

### Component usage

```js
<LightTheme>
  <Box ref={ref}>
    <ContentDrop
      anchorEl={ref}
      items={[
        {
          color: 'primary',
          description: 'This is an example text',
          name: 'This is an example name',
          href: '/home',
          icon: <InfoIcon />,
        },
      ]}
      open
    />
  </Box>
</LightTheme>
```

### Render a custom popover menu

```js live
<LightTheme>
  <ContentDrop
    items={[
      {
        color: 'primary',
        description: 'This is an example text',
        name: 'This is an example name',
        href: '/home',
        icon: <InfoIcon />,
      },
    ]}
    open
  />
</LightTheme>
```
