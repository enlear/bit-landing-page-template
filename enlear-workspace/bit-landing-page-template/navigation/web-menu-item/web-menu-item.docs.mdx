import { WebMenuItem } from './web-menu-item';

A component that renders a menu item

### Component usage

```js
<WebMenuItem
  item={{
    color: 'primary',
    label: 'Menu Item',
    expandable: true,
    href: '/menu-item',
    sub: [],
  }}
/>
```
